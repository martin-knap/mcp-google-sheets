  Proposed Architecture: 6 Tools + Presets

  Tool 1: sheets_data - Read/Write/Search

  sheets_data(
      spreadsheet_id: str,
      sheet: str,
      action: "read" | "write" | "clear" | "search" | "replace" | "sort",
      range: str = None,

      # Read options
      include_formulas: bool = False,

      # Write options
      data: list = None,
      style: str = None,              # "table", "table_striped", etc.
      column_types: dict = None,      # {"B": "currency", "C": "date", "D": "percent"}
      auto_format: bool = False,      # Auto-detect types

      # Search options (action="search")
      filters: list = None,           # [{"column": "A", "op": "contains", "value": "x"}]
      match_all: bool = True,         # AND vs OR

      # Replace options (action="replace")
      find: str = None,
      replace_with: str = None,
      use_regex: bool = False,

      # Sort options (action="sort")
      sort_by: list = None,           # [{"column": "B", "order": "desc"}]
  )

  Tool 2: sheets_format - All Formatting

  sheets_format(
      spreadsheet_id: str,
      sheet: str,
      action: "style" | "border" | "merge" | "unmerge" | "conditional" | "clear",
      range: str,

      # Style presets (action="style")
      style: str = None,              # "h1", "h2", "bold", "muted", "error", etc.

      # Or explicit format
      bold: bool = None,
      italic: bool = None,
      font_size: int = None,
      font_color: str = None,
      bg_color: str = None,
      align: str = None,              # "left", "center", "right"
      valign: str = None,             # "top", "middle", "bottom"
      wrap: bool = None,
      number_format: str = None,      # Or preset: "currency", "percent", "date"

      # Border options (action="border")
      border_style: str = "solid",
      border_color: str = "black",
      border_sides: str = "all",      # "all", "outer", "inner", "bottom", etc.

      # Conditional format (action="conditional") - NEW
      rule: str = None,               # "color_scale", "data_bar", "icon_set", "custom"
      condition: dict = None,         # {"type": "greater_than", "value": 100, "format": {"bg_color": "green"}}

      # Batch mode - apply multiple formats at once
      formats: list = None,           # [{"range": "A1:B1", "style": "h1"}, ...]
  )

  Tool 3: sheets_structure - Rows, Cols, Size, Validation

  sheets_structure(
      spreadsheet_id: str,
      sheet: str,
      action: "resize" | "freeze" | "add_rows" | "add_cols" | "delete_rows" | "delete_cols" | "validate" | "group" | "protect",

      # Resize options
      columns: str = None,            # "A:D" or "all"
      rows: str = None,               # "1:10" or "all"
      width: int = None,              # Explicit width in pixels
      height: int = None,             # Explicit height in pixels
      auto: bool = True,              # Auto-fit content

      # Freeze options
      freeze_rows: int = 0,
      freeze_cols: int = 0,

      # Add/Delete options
      start: int | str = None,        # Row number or column letter
      count: int = 1,

      # Validation options (action="validate")
      range: str = None,
      validation: str = None,         # "dropdown", "number", "date", "checkbox", "custom"
      options: list = None,           # For dropdown
      min_value: float = None,        # For number
      max_value: float = None,
      custom_formula: str = None,

      # Protect options (action="protect") - NEW
      protected_range: str = None,
      warning_only: bool = False,
  )

  Tool 4: sheets_visualize - Charts, Pivots, Sparklines (NEW)

  sheets_visualize(
      spreadsheet_id: str,
      sheet: str,
      action: "chart" | "pivot" | "sparkline",

      # Chart options
      chart_type: str = None,         # "line", "bar", "column", "pie", "area", "scatter", "combo"
      data_range: str = None,
      title: str = None,
      position: str = None,           # "A10" - anchor cell
      size: tuple = None,             # (width, height) in pixels

      # Preset chart styles
      style: str = None,              # "default", "minimal", "colorful"

      # Pivot options
      source_range: str = None,
      rows: list = None,              # ["Category", "Product"]
      columns: list = None,           # ["Region"]
      values: list = None,            # [{"field": "Sales", "summarize": "sum"}]
      filters: list = None,

      # Sparkline (creates formula in cells)
      sparkline_type: str = None,     # "line", "bar", "column"
      sparkline_range: str = None,
      target_cell: str = None,
  )

  Tool 5: sheets_manage - Sheet Tabs

  sheets_manage(
      spreadsheet_id: str,
      action: "list" | "create" | "rename" | "copy" | "delete" | "duplicate",
      sheet: str = None,
      new_name: str = None,
      destination_spreadsheet: str = None,

      # Create options
      rows: int = 1000,
      cols: int = 26,
      tab_color: str = None,
  )

  Tool 6: drive - Spreadsheets & Folders

  drive(
      action: "create" | "list" | "share" | "folders" | "info" | "summary",

      spreadsheet_id: str = None,
      title: str = None,
      folder_id: str = None,

      # Share options
      recipients: list = None,        # [{"email": "x@y.com", "role": "writer"}]

      # Summary options
      spreadsheet_ids: list = None,
      preview_rows: int = 5,
  )
